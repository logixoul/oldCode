namespace stefanfw {

	ci::app::WindowRef textWindow;

	static std::stringstream redirectStream;
	
	void createTextWindow(ci::app::App& app) {
		textWindow = app.createWindow(Window::Format().size(400, 400));
		std::cout.rdbuf(redirectStream.rdbuf());
	}

	bool shouldDrawTextWindow(ci::app::App& app) {
		return app.getWindow() == textWindow;
	}

	void drawTextWindow() {
		gl::clear(Color(0, 0, 0));
		gl::drawString(redirectStream.str(), vec2());
		redirectStream.clear();
	}
